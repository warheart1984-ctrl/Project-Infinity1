"""
AAIS Multi-LLM Integration Module
===================================
Governed multi-provider LLM orchestration for the AAIS platform.

Supports 12 providers via unified lane/contract architecture:
  - OpenAI (GPT-4.1, GPT-5.4, o3, o4-mini)
  - Azure OpenAI
  - Anthropic Claude (sonnet-4-5, opus-4)
  - Google Gemini (2.5-flash, 3-flash-preview)
  - xAI Grok (grok-3)
  - DeepSeek (v4-flash, v4-pro)
  - Mistral (medium-latest, large-latest)
  - Cohere (command-r-plus, command-a-03-2025)
  - Together AI (Llama, Qwen, etc.)
  - Groq (ultra-fast inference)
  - Perplexity (search-grounded)
  - AWS Bedrock (multi-model)

Install all providers:
  pip install openai anthropic google-genai mistralai cohere together groq boto3

Architecture:
  Orchestrator
  ├── LaneRegistry   → provider configs + model routing
  ├── LaneContract   → system prompt, role, constraints per lane
  ├── LawEngine      → output validation (phrase caps, word bounds, drift)
  └── Provider impls → unified call() interface per provider

Author: Jon Halstead / AAIS Platform
"""

import os

# ═══════════════════════════════════════════════════════════════════
#  SECTION 1 — DATA STRUCTURES
# ═══════════════════════════════════════════════════════════════════

class ProviderType(Enum):
    OPENAI          = "openai"
    AZURE_OPENAI    = "azure_openai"
    ANTHROPIC       = "anthropic"
    GOOGLE_GEMINI   = "google_gemini"
    XAI_GROK        = "xai_grok"
    DEEPSEEK        = "deepseek"
    MISTRAL         = "mistral"
    COHERE          = "cohere"
    TOGETHER        = "together"
    GROQ            = "groq"
    PERPLEXITY      = "perplexity"
    AWS_BEDROCK     = "aws_bedrock"


@dataclass
class LaneContract:
    """Defines the behavioral contract for a governed LLM lane."""
    name: str                              # lane identifier
    provider: ProviderType                 # which provider to call
    model: str                             # model ID
    role: str                              # human-readable role description
    system_prompt: str                     # system instructions for this lane
    max_tokens: int = 4096                 # output cap
    temperature: float = 0.7              # creativity dial
    phrase_caps: dict = field(default_factory=dict)   # {"phrase": max_count}
    min_words: int = 0                     # minimum output length
    max_words: int = 999999                # maximum output length
    retry_on_violation: bool = True        # auto-retry if law engine rejects
    max_retries: int = 2                   # retry cap


@dataclass
class LaneResult:
    """Output from a governed lane execution."""
    lane_name: str
    provider: str
    model: str
    text: str
    word_count: int
    valid: bool
    violations: list
    latency_ms: float
    input_tokens: int = 0
    output_tokens: int = 0

class BaseProvider(ABC):
    """Abstract base for all LLM providers."""

    @abstractmethod
    def call(self, model: str, system: str, prompt: str,
             max_tokens: int, temperature: float) -> dict:
        """
        Returns: {"text": str, "input_tokens": int, "output_tokens": int}
        """
        pass


# ─────────────────────────────────────────────────────────────────
#  2A — OpenAI  (GPT-4.1, GPT-5.4, o3, o4-mini)
# ─────────────────────────────────────────────────────────────────
class OpenAIProvider(BaseProvider):
    """
    Direct OpenAI API.
    Env: OPENAI_API_KEY
    pip install openai
    Models: gpt-4.1, gpt-4.1-mini, gpt-4.1-nano, gpt-5.4, gpt-5.4-mini,
            o3, o3-mini, o4-mini
    """
    def __init__(self):
        from openai import OpenAI
        self.client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

    def call(self, model, system, prompt, max_tokens, temperature):
        resp = self.client.chat.completions.create(
            model=model,
            max_tokens=max_tokens,
            temperature=temperature,
            messages=[
                {"role": "system", "content": system},
                {"role": "user", "content": prompt}
            ]
        )
        choice = resp.choices[0]
        usage = resp.usage or {}
        return {
            "text": choice.message.content,
            "input_tokens": getattr(usage, "prompt_tokens", 0),
            "output_tokens": getattr(usage, "completion_tokens", 0),
        }

# ─────────────────────────────────────────────────────────────────
#  2D — Google Gemini
# ─────────────────────────────────────────────────────────────────
class GeminiProvider(BaseProvider):
    """
    Google Gemini API.
    Env: GEMINI_API_KEY
    pip install google-genai
    Models: gemini-2.5-flash, gemini-2.5-pro, gemini-3-flash-preview
    """
    def __init__(self):
        from google import genai
        self.client = genai.Client(api_key=os.environ["GEMINI_API_KEY"])

    def call(self, model, system, prompt, max_tokens, temperature):
        from google.genai import types
        response = self.client.models.generate_content(
            model=model,
            contents=prompt,
            config=types.GenerateContentConfig(
                system_instruction=system,
                max_output_tokens=max_tokens,
                temperature=temperature,
            )
        )
        # Token counts from usage_metadata
        usage = response.usage_metadata
        return {
            "text": response.text,
            "input_tokens": getattr(usage, "prompt_token_count", 0),
            "output_tokens": getattr(usage, "candidates_token_count", 0),
        }

#  2L — AWS Bedrock  (Claude, Llama, Mistral, Titan on AWS)
# ─────────────────────────────────────────────────────────────────
class BedrockProvider(BaseProvider):
    """
    AWS Bedrock — multi-model access via AWS.
    Env: AWS_ACCESS_KEY_ID, AWS_SECRET_ACCESS_KEY, AWS_DEFAULT_REGION
    pip install boto3
    Models: anthropic.claude-sonnet-4-5-20250514-v1:0,
            meta.llama3-3-70b-instruct-v1:0,
            mistral.mistral-large-2411-v1:0,
            amazon.titan-text-premier-v2:0
    """
    def __init__(self):
        import boto3
        self.client = boto3.client(
            "bedrock-runtime",
            region_name=os.environ.get("AWS_DEFAULT_REGION", "us-east-1"),
        )

    def call(self, model, system, prompt, max_tokens, temperature):
        body = json.dumps({
            "anthropic_version": "bedrock-2023-05-31",
            "max_tokens": max_tokens,
            "temperature": temperature,
            "system": system,
            "messages": [{"role": "user", "content": prompt}]
        })
        resp = self.client.invoke_model(
            modelId=model,
            contentType="application/json",
            accept="application/json",
            body=body,
        )
        result = json.loads(resp["body"].read())
        return {
            "text": result["content"][0]["text"],
            "input_tokens": result.get("usage", {}).get("input_tokens", 0),
            "output_tokens": result.get("usage", {}).get("output_tokens", 0),
        }

# ═══════════════════════════════════════════════════════════════════
#  SECTION 4 — LAW ENGINE  (output validation & governance)
# ═══════════════════════════════════════════════════════════════════

class LawEngine:
    """
    Validates LLM output against contract rules.
    Enforces phrase caps, word count bounds, pattern bans.
    """

    @staticmethod
    def validate(text: str, contract: LaneContract) -> dict:
        violations = []

        # ── Phrase cap enforcement ──
        for phrase, max_count in contract.phrase_caps.items():
            count = len(re.findall(re.escape(phrase), text, re.IGNORECASE))
            if count > max_count:
                violations.append(
                    f"PHRASE_CAP: '{phrase}' appears {count}x "
                    f"(max {max_count})"
                )

        # ── Word count bounds ──
        wc = len(text.split())
        if wc < contract.min_words:
            violations.append(
                f"UNDERCOUNT: {wc} words (min {contract.min_words})"
            )
        if wc > contract.max_words:
            violations.append(
                f"OVERCOUNT: {wc} words (max {contract.max_words})"
            )

        # ── Drift detection: meta-commentary leak ──
        drift_patterns = [
            r"as an ai",
            r"i cannot",
            r"i'm happy to",
            r"here(?:'s| is) (?:the|a|an|your)",
            r"certainly!",
            r"of course!",
        ]
        for pat in drift_patterns:
            if re.search(pat, text, re.IGNORECASE):
                violations.append(f"DRIFT: meta-commentary detected: '{pat}'")

        return {
            "valid": len(violations) == 0,
            "violations": violations,
            "word_count": wc,
        }


# ═══════════════════════════════════════════════════════════════════
#  SECTION 5 — ORCHESTRATOR  (the governed pipeline)
# ═══════════════════════════════════════════════════════════════════

class Orchestrator:
    """
    Governed multi-LLM orchestrator for AAIS.

    Usage:
        orch = Orchestrator()
        orch.register_lane(my_contract)
        result = orch.run("lane_name", "my prompt")
    """

    def __init__(self):
        self._lanes: dict[str, LaneContract] = {}
        self._providers: dict[ProviderType, BaseProvider] = {}

    def register_lane(self, contract: LaneContract):
        """Register a governed lane with its contract."""
        self._lanes[contract.name] = contract
        # Lazy-init provider if not already loaded
        if contract.provider not in self._providers:
            provider_cls = PROVIDER_MAP[contract.provider]
            self._providers[contract.provider] = provider_cls()
        logger.info(f"Lane registered: {contract.name} "
                     f"→ {contract.provider.value}/{contract.model}")

    def run(self, lane_name: str, prompt: str) -> LaneResult:
        """Execute a single governed lane call with law engine validation."""
        contract = self._lanes[lane_name]
        provider = self._providers[contract.provider]

        for attempt in range(1, contract.max_retries + 1):
            t0 = time.time()
            try:
                raw = provider.call(
                    model=contract.model,
                    system=contract.system_prompt,
                    prompt=prompt,
                    max_tokens=contract.max_tokens,
                    temperature=contract.temperature,
                )
            except Exception as e:
                logger.error(f"Lane {lane_name} call failed: {e}")
                raise

            latency = (time.time() - t0) * 1000
            text = raw["text"]

            # ── Law engine validation ──
            check = LawEngine.validate(text, contract)

            if check["valid"] or not contract.retry_on_violation:
                return LaneResult(
                    lane_name=lane_name,
                    provider=contract.provider.value,
                    model=contract.model,
                    text=text,
                    word_count=check["word_count"],
                    valid=check["valid"],
                    violations=check["violations"],
                    latency_ms=round(latency, 1),
                    input_tokens=raw.get("input_tokens", 0),
                    output_tokens=raw.get("output_tokens", 0),
                )

            logger.warning(
                f"Lane {lane_name} attempt {attempt} violations: "
                f"{check['violations']} — retrying"
            )
            # Append correction instruction for retry
            prompt = (
                f"{prompt}\n\n[CORRECTION REQUIRED — previous output "
                f"violated: {', '.join(check['violations'])}. "
                f"Fix these issues.]"
            )

        # Return last attempt even if still violating
        return LaneResult(
            lane_name=lane_name,
            provider=contract.provider.value,
            model=contract.model,
            text=text,
            word_count=check["word_count"],
            valid=check["valid"],
            violations=check["violations"],
            latency_ms=round(latency, 1),
            input_tokens=raw.get("input_tokens", 0),
            output_tokens=raw.get("output_tokens", 0),
        )

    def pipeline(self, lane_sequence: list[str], initial_prompt: str,
                 pass_output_forward: bool = True) -> list[LaneResult]:
        """
        Run a multi-lane pipeline (e.g., spine → voltage → clarity).
        Each lane receives the previous lane's output as its prompt
        if pass_output_forward is True.
        """
        results = []
        current_prompt = initial_prompt

        for lane_name in lane_sequence:
            result = self.run(lane_name, current_prompt)
            results.append(result)
            if pass_output_forward:
                current_prompt = result.text

        return results

    def list_lanes(self) -> dict:
        """List all registered lanes and their configs."""
        return {
            name: {
                "provider": c.provider.value,
                "model": c.model,
                "role": c.role,
            }
            for name, c in self._lanes.items()
        }

def example_narrative_merge():
    """
    Example: Veilthorn-style multi-voice merge pipeline.

    """
    orch = Orchestrator()

    # Register the narrative lanes
    orch.register_lane(NARRATIVE_LANES["spine"])
    orch.register_lane(NARRATIVE_LANES["voltage"])
    orch.register_lane(NARRATIVE_LANES["clarity"])

    # Source chapter text (from your draft)
    source_chapter = """
    [Paste your raw chapter text here]
    """

    # Run the governed pipeline
    results = orch.pipeline(
        lane_sequence=["spine", "voltage", "clarity"],
        initial_prompt=f"Rewrite this chapter:\n\n{source_chapter}",
        pass_output_forward=True,

    # Print results
    for r in results:

              f"Latency: {r.latency_ms}ms")

            print(f"Violations: {r.violations}")
        print(f"{'='*60}")
        print(r.text[:500] + "...")
