# src/continuity/comprehension_fitness.py
from dataclasses import dataclass
from typing import Dict

@dataclass
class ComprehensionComponents:
  C_loc: float   # locality
  C_clr: float   # clarity
  C_cons: float  # consequence visibility
  C_link: float  # linkage

@dataclass
class ComprehensionConfig:
  v_loc: float = 0.25
  v_clr: float = 0.25
  v_cons: float = 0.25
  v_link: float = 0.25
  theta_min: float = 0.75
  delta_max: float = 0.10

def clamp01(x: float) -> float:
  return max(0.0, min(1.0, x))

def compute_chi(components: ComprehensionComponents,
                cfg: ComprehensionConfig) -> float:
  chi = (
    cfg.v_loc * components.C_loc +
    cfg.v_clr * components.C_clr +
    cfg.v_cons * components.C_cons +
    cfg.v_link * components.C_link
  )
  return clamp01(chi)

def compute_law_comprehension(law_record: Dict,
                              graph_metrics: Dict,
                              cfg: ComprehensionConfig) -> float:
  """
  law_record: includes purpose_text, consequences_text, links to evidence/epochs
  graph_metrics: includes hop_count, fan_in, fan_out, etc.
  """
  # locality: fewer hops + bounded fan-in/out
  C_loc = clamp01(1.0 / (1.0 + graph_metrics["avg_hops"]))
  # clarity: inverse of ambiguity / conflicting annotations
  C_clr = clamp01(1.0 - graph_metrics["ambiguity_score"])
  # consequence visibility: presence + quality of "what breaks if removed"
  C_cons = clamp01(graph_metrics["consequence_coverage"])
  # linkage: cross-ledger links density (law↔evidence↔epoch)
  C_link = clamp01(graph_metrics["link_density"])

  comps = ComprehensionComponents(
    C_loc=C_loc,
    C_clr=C_clr,
    C_cons=C_cons,
    C_link=C_link,
  )
  return compute_chi(comps, cfg)

# src/continuity/comprehension_ledger.py
import hashlib
from typing import Optional
from .comprehension_fitness import ComprehensionComponents, ComprehensionConfig, compute_chi

class ComprehensionLedgerStore:
    def __init__(self, path: str):
        self.conn = sqlite3.connect(path)
        self.conn.row_factory = sqlite3.Row

    def _hash_payload(self, payload: dict, prev_hash: Optional[str]) -> str:
        h = hashlib.sha256()
        h.update(json.dumps(payload, sort_keys=True).encode("utf-8"))
        if prev_hash:
            h.update(prev_hash.encode("utf-8"))
        return h.hexdigest()

    def get_last_hash(self) -> Optional[str]:
        cur = self.conn.execute(
            "SELECT hash FROM comprehension_ledger ORDER BY created_at DESC LIMIT 1"
        )
        row = cur.fetchone()
        return row["hash"] if row else None

    def record_comprehension(self, object_type: str, object_id: str,
                             epoch: int, comps: ComprehensionComponents,
                             cfg: ComprehensionConfig, prev_chi: Optional[float]) -> float:
        chi = compute_chi(comps, cfg)
        now = time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime())

        rec_id = f"CHI-{object_type}-{object_id}-E{epoch}"
        self.conn.execute(
            """
            INSERT OR REPLACE INTO comprehension_records
            (id, object_type, object_id, chi, C_loc, C_clr, C_cons, C_link, epoch, created_at)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
            """,
            (rec_id, object_type, object_id, chi,
             comps.C_loc, comps.C_clr, comps.C_cons, comps.C_link,
             epoch, now)
        )

        delta = None
        if prev_chi is not None:
            delta = chi - prev_chi

        payload = {
            "chi": chi,
            "components": {
                "C_loc": comps.C_loc,
                "C_clr": comps.C_clr,
                "C_cons": comps.C_cons,
                "C_link": comps.C_link,
            },
            "thresholds": {
                "theta_min": cfg.theta_min,
                "delta_max": cfg.delta_max,
            },
            "prev_chi": prev_chi,
            "delta": delta,
        }

        prev_hash = self.get_last_hash()
        entry_hash = self._hash_payload(payload, prev_hash)
        entry_id = f"CHI-LEDGER-{object_type}-{object_id}-E{epoch}"

        status = "ok"
        if chi < cfg.theta_min:
            status = "breach"
        elif delta is not None and abs(delta) > cfg.delta_max:
            status = "warning"
        payload["status"] = status

        self.conn.execute(
            """
            INSERT INTO comprehension_ledger
            (entry_id, epoch, entry_type, object_type, object_id,
             payload_json, prev_hash, hash, created_at)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
            """,
            (entry_id, epoch, "CHI_EVAL", object_type, object_id,
             json.dumps(payload), prev_hash, entry_hash, now)
        )
        self.conn.commit()
        return chi

from typing import Optional
from .comprehension_fitness import ComprehensionComponents, ComprehensionConfig, compute_chi

class ComprehensionLedgerStore:
    def __init__(self, path: str):
        self.conn = sqlite3.connect(path)
        self.conn.row_factory = sqlite3.Row

    def _hash_payload(self, payload: dict, prev_hash: Optional[str]) -> str:
        h = hashlib.sha256()
        h.update(json.dumps(payload, sort_keys=True).encode("utf-8"))
        if prev_hash:
            h.update(prev_hash.encode("utf-8"))
        return h.hexdigest()

    def get_last_hash(self) -> Optional[str]:
        cur = self.conn.execute(
            "SELECT hash FROM comprehension_ledger ORDER BY created_at DESC LIMIT 1"
        )
        row = cur.fetchone()
        return row["hash"] if row else None

    def record_comprehension(self, object_type: str, object_id: str,
                             epoch: int, comps: ComprehensionComponents,
                             cfg: ComprehensionConfig, prev_chi: Optional[float]) -> float:
        chi = compute_chi(comps, cfg)
        now = time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime())

        rec_id = f"CHI-{object_type}-{object_id}-E{epoch}"
        self.conn.execute(
            """
            INSERT OR REPLACE INTO comprehension_records
            (id, object_type, object_id, chi, C_loc, C_clr, C_cons, C_link, epoch, created_at)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
            """,
            (rec_id, object_type, object_id, chi,
             comps.C_loc, comps.C_clr, comps.C_cons, comps.C_link,
             epoch, now)
        )

        delta = None
        if prev_chi is not None:
            delta = chi - prev_chi

        payload = {
            "chi": chi,
            "components": {
                "C_loc": comps.C_loc,
                "C_clr": comps.C_clr,
                "C_cons": comps.C_cons,
                "C_link": comps.C_link,
            },
            "thresholds": {
                "theta_min": cfg.theta_min,
                "delta_max": cfg.delta_max,
            },
            "prev_chi": prev_chi,
            "delta": delta,
        }

        prev_hash = self.get_last_hash()
        entry_hash = self._hash_payload(payload, prev_hash)
        entry_id = f"CHI-LEDGER-{object_type}-{object_id}-E{epoch}"

        status = "ok"
        if chi < cfg.theta_min:
            status = "breach"
        elif delta is not None and abs(delta) > cfg.delta_max:
            status = "warning"
        payload["status"] = status

        self.conn.execute(
            """
            INSERT INTO comprehension_ledger
            (entry_id, epoch, entry_type, object_type, object_id,
             payload_json, prev_hash, hash, created_at)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
            """,
            (entry_id, epoch, "CHI_EVAL", object_type, object_id,
             json.dumps(payload), prev_hash, entry_hash, now)
        )
        self.conn.commit()
        return chi
